Javascript ES6 Cheat Sheet

Arrow Function

const sum = (a, b) => a + b

console.log(sum(2, 6)) // prints 8

Object Property Assignment

const a = 2

const b = 5

const obj = { a, b }

// Before es6:

// obj = { a: a, b: b }

console.log(obj)

// prints = { a: 2, b: 5 }

Object Function Assignment

const obj = {

a: 5,

b() {

console.log('b')

  }

}

obj.b() // prints "b"
